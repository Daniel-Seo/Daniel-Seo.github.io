append 와 after 모두 jQuery 에서 동적으로 html tag 를 넣을 때 사용한다.

1. append

<div id="target">
 <div>already exist div</div>
</div>

<script>
$('#target').append('<div>appended div</div>');
</script>

<div id="target">
 <div>already exist div</div>
 <div>appended div</div>
</div>

2. after

<div id="target">
 <div>already exist div</div>
</div>

<script>
$('#target').after('<div>after div</div>');
</script>

<div id="target">
 <div>already exist div</div>
</div>
<div>after div</div>
